// memory_allocation test_main: used 4, spilled 0

fun test_main() {
    int i; i = 0;
    int j; j = 0;
    int count; count = 0;
    while (i < 4) {
        j = 0;
        while (j < 4) {
            j = j + 1;
            if (j == 2) continue;
            if (i == 3) break;
            count = count + 1;
        }
        i = i + 1;
    }
    assertEquals(count, 9)
}
