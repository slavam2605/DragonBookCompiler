// #allow_undef

int i = 0;
int sum = 0;
for (; ; i = i + 1) {
    if (i == 7) {
        break;
    }
    if ((i % 2) == 1) {
        continue;
    }
    sum = sum + i;
}
assertEquals(sum, 0 + 2 + 4 + 6);

bool b = undef;
bool guard = (b && false) || (true && b) || (false && true);
assertStaticUnknown(guard);

int t = 0;
if (true) {
    if (false) {
        t = 100;
    } else {
        t = 3;
    }
} else {
    t = 4;
}
assertStaticEquals(t, 3);

int keep = 9;
while (false) { keep = 0; }
assertStaticEquals(keep, 9);
