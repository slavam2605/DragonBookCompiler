fun test_main() {
    int a = 17
    int b = 5
    int result_add = a + b
    int result_sub = a - b
    int result_mul = a * b
    int result_div = a / b
    int result_mod = a % b
    int result_neg_a = -a;
    int result_neg_b = -b;
    assertStaticEquals(result_add, 22)
    assertStaticEquals(result_sub, 12)
    assertStaticEquals(result_mul, 85)
    assertStaticEquals(result_div, 3)
    assertStaticEquals(result_mod, 2)
    assertStaticEquals(result_neg_a, -17)
    assertStaticEquals(result_neg_b, -5)
}
