int a = undef;                   assertStaticUnknown(a)
bool b = undef;                  assertStaticUnknown(b)

int result1 = a - a;             assertStaticEquals(result1, 0)
int result2 = a % 1;             assertStaticEquals(result2, 0)
int result3 = a * 0;             assertStaticEquals(result3, 0)
int result4 = 0 * a;             assertStaticEquals(result4, 0)
bool result5 = a == a;           assertStaticEquals(result5, 1)
bool result6 = a != a;           assertStaticEquals(result6, 0)
bool result7 = a < a;            assertStaticEquals(result7, 0)
bool result8 = a <= a;           assertStaticEquals(result8, 1)
bool result9 = a >= a;           assertStaticEquals(result9, 1)
bool result10 = a > a;           assertStaticEquals(result10, 0)
bool result11 = true || b;       assertStaticEquals(result11, 1)
bool result12 = false && b;      assertStaticEquals(result12, 0)
