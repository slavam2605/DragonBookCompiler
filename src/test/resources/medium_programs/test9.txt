fun test_main() {
    int i = 0;
    int cnt = 0;
    for (; i < 10; ) {
        i = i + 1;
        if ((i % 3) == 0) {
            continue;
        }
        if (i > 7) {
            break;
        }
        cnt = cnt + 1;
    }
    assertEquals(cnt, 5);

    int c = 0;
    if (false) {
        c = 1;
    } else if (true) {
        c = 2;
    } else {
        c = 3;
    }
    assertStaticEquals(c, 2);

    bool b = undef(false);
    bool d = (b && true) || (false && b);
    assertStaticUnknown(d);

    int keep = 7;
    if (false) { keep = 0; }
    assertStaticEquals(keep, 7);
}
