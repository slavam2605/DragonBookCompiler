fun test_main() {
    bool b = false
    int i = 0
    while (i < 1) {
        b = !b
        i = i + 1
    }

    bool result_b = b
    bool result1 = b || true
    bool result2 = b && false
    assertStaticUnknown(result_b)
    assertStaticEquals(result1, true)
    assertStaticEquals(result2, false)
}
