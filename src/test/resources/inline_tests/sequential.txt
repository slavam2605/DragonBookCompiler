// Test multiple different functions inlined sequentially in same block
fun double(int x) -> int {
    return x * 2
}

fun triple(int x) -> int {
    return x * 3
}

fun increment(int x) -> int {
    return x + 1
}

fun negate(int x) -> int {
    return -x
}

fun test_main() {
    // Multiple different function calls in sequence
    int a = double(5)
    int b = triple(4)
    int c = increment(a)
    int d = negate(b)

    assertStaticEquals(a, 10)   // 5 * 2
    assertStaticEquals(b, 12)   // 4 * 3
    assertStaticEquals(c, 11)   // 10 + 1
    assertStaticEquals(d, -12)  // -12

    // Combined in single expression
    int result = double(3) + triple(2) + increment(1)
    assertStaticEquals(result, 14)  // 6 + 6 + 2 = 14
}
