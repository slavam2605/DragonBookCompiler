fun test_main() {
    float a = 17.5
    float b = 5.25
    bool result_eq = a == b
    bool result_nq = a != b
    bool result_lt = a < b
    bool result_le = a <= b
    bool result_gt = a > b
    bool result_ge = a >= b
    assertStaticEquals(result_eq, 0)
    assertStaticEquals(result_nq, 1)
    assertStaticEquals(result_lt, 0)
    assertStaticEquals(result_le, 0)
    assertStaticEquals(result_gt, 1)
    assertStaticEquals(result_ge, 1)

    // Test equality with same values
    float x = 10.5
    float y = 10.5
    bool result_eq2 = x == y
    bool result_nq2 = x != y
    assertStaticEquals(result_eq2, 1)
    assertStaticEquals(result_nq2, 0)
}
