int i = 0;
int cnt = 0;
for (; i < 10; ) {
    i = i + 1;
    if ((i % 3) == 0) {
        continue;
    }
    if (i > 7) {
        break;
    }
    cnt = cnt + 1;
}
assertEquals(cnt, 5);
