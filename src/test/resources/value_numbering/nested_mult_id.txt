fun test_main() {
    int a = undef(0)
    int b = undef(0)
    int t1 = (a * 0) + b
    assertStaticEquals(t1 == b, true)

    int t2 = (1 * a) + (b * 0)
    assertStaticEquals(t2 == a, true)

    float x = undef(0.0)
    float y = undef(0.0)
    float t3 = (x * 0.0) + y
    assertStaticEquals(t3 == y, true)

    float t4 = (1.0 * x) + (y * 0.0)
    assertStaticEquals(t4 == x, true)
}
