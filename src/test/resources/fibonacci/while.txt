fun test_main() -> int {
    int prev = 0
    int curr = 1
    int n = <$n>
    int i = 0
    int r
    if (n == 0) {
        r = 0
    } else if (n == 1) {
        r = 1
    } else if (n == 2) {
        r = 1
    } else {
        while (i < n) {
            int temp = curr
            curr = prev + curr
            prev = temp
            i = i + 1
        }
        r = prev
    }
    return r
}
