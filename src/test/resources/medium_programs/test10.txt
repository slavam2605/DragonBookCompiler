// memory_allocation test_main: used 3, spilled 0

fun test_main() {
    for (; ; ) { break; }

    int n = 5;
    int fact = 1;
    while (n > 0) {
        fact = fact * n;
        n = n - 1;
    }
    assertEquals(fact, 120);

    int x = undef(0);
    int y = x * 0;
    assertStaticEquals(y, 0);

    bool b1 = undef(false);
    bool b2 = !b1;
    assertStaticUnknown(b2);

    int choose;
    if (true) {
        if (false) {
            choose = 9;
        } else {
            choose = 8;
        }
    } else {
        choose = 7;
    }
    assertStaticEquals(choose, 8);

    int j = 0;
    int odd = 0;
    for (; j < 6; j = j + 1) {
        if ((j % 2) == 0) { continue; }
        odd = odd + j;
    }
    assertEquals(odd, 1 + 3 + 5);
}
