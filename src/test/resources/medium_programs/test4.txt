int count = 0;
for (int i = 0; i < 5; i = i + 1) {
    for (int j = 0; j < 5; j = j + 1) {
        if (i == j) {
            continue;
        }
        if (i + j > 6) {
            break;
        }
        count = count + 1;
    }
}
assertEquals(count, 18);

bool always = true || (count == 0);
bool never = false && (count == 18);
assertStaticEquals(always, true);
assertStaticEquals(never, false);

int a = 10;
while (false) { a = 0; }
assertStaticEquals(a, 10);

int k = 0;
while (k < 3) {
    k = k + 1;
}
assertEquals(k, 3);
