fun test_main() {
    int a = 17
    int b = 5
    bool result_eq = a == b
    bool result_nq = a != b
    bool result_lt = a < b
    bool result_le = a <= b
    bool result_gt = a > b
    bool result_ge = a >= b
    assertStaticEquals(result_eq, 0)
    assertStaticEquals(result_nq, 1)
    assertStaticEquals(result_lt, 0)
    assertStaticEquals(result_le, 0)
    assertStaticEquals(result_gt, 1)
    assertStaticEquals(result_ge, 1)
}
